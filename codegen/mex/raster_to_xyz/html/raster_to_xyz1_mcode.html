1%% raster to grid xyz yearly 23 data
 2% Input parameter (raster_file, fil, output)
 3% raster_file *.tif (geotiff) multi band file
 4% fil --> filter (wavelet or lowpass) default = no filter
 5% output --> output format (*.txt or *.nc)
 6%%
 7function raster_to_xyz(raster_file, fil, output)
 8%raster_file = [part,list(1).name];
 9%raster_file = '/mnt/RAM_disk/EVI_Java_2010001_2016353.tif';
10%raster_file = '/home/alvin/TIFF/jawa/EVI_Java_2010001_2016353.tif'
11[raster, R] = geotiffread(raster_file);
12[x,y,z] = size(raster);
13raster = single(raster);
14%mask = data;
15%pola = data{n};
16ukuran_block=[25 25];
17
18%normalisasi data sehingga mendapat NaN
19fun_normalisasi=@(block_struct)map_reduce(block_struct.data);
20datanorm=blockproc(raster,ukuran_block,fun_normalisasi,'useparallel',true);
21datanorm=reshape(single(datanorm),[x*y 1]);
22%datanorm=~isnan(datanorm(:,1));
23
24    switch fil
25        case 'lowpass'
26            fun_filter=@(block_struct)lowpass(block_struct.data);
27            raster=blockproc(raster,ukuran_block,fun_filter,'useparallel',true);
28            raster_file = [raster_file,'_lowpass',];
29        case 'wavelet'
30            fun_filter=@(block_struct)transformasi_coiflet1_lvl1(block_struct.data);
31            raster=blockproc(raster,ukuran_block,fun_filter,'useparallel',true);
32            raster_file = [raster_file,'_wavelet'];
33        otherwise
34            disp('No Filter');    
35
36    end
37
38raster = reshape(raster,[x*y z]);
39raster = raster(~isnan(datanorm(:,1)),:);
40[col,row] = size(raster);
41img(1:col*row/23,1) = int16(0);
42
43for i = 1:row/23
44    img((i)*col,1)=1;
45    img(1+(i-1)*col:(i)*col,2:23+1) = raster(:,1+(i-1)*23:23*i);
46end
47
48[cell, band] = size(img);
49band = band-1;
50clear raster;
51
52%nccreate([raster_file,'.nc'],'x','y','band', 'data');
53    switch output 
54        case 'txt'
55        fName = [raster_file,'.txt'];         %# A file name
56        fid = fopen(fName,'w');            %# Open the file
57        dlmwrite(fName,img,'-append',...  %# Print the matrix
58                 'delimiter','\t',...
59                 'newline','pc');
60
61        fclose(fid);
62
63        case 'nc'
64        %create netCDF
65         nccreate([raster_file,'.nc'],'Data',...
66              'Dimensions',{'num_coordinates' band 'num_elements' cell},...
67              'Format','netcdf4', 'Datatype', 'single');
68         ncwrite([raster_file,'.nc'],'Data', single(img(:,2:end)'));
69         %ncdisp([raster_file,'.nc']);
70        otherwise
71            disp('Check output parameter');
72    end
73end
