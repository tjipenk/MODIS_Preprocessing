1%% raster to grid xyz yearly 23 data
 2function xyz_to_raster(input_file ,raster_file, output)
 3
 4%raster_file = '/mnt/RAM_disk/Sawah_2015.tif';
 5%input_file = '/mnt/RAM_disk/Sawah_2015.tif_lowpass.txt.membership';
 6[raster, R] = geotiffread(raster_file);
 7[x,y,z] = size(raster);
 8
 9
10%mask = data;
11%pola = data{n};
12ukuran_block=[25 25];
13
14%normalisasi data sehingga mendapat NaN
15fun_normalisasi=@(block_struct)map_reduce(block_struct.data);
16datanorm=blockproc(raster,ukuran_block,fun_normalisasi,'useparallel',true);
17datanorm=reshape(single(datanorm),[x*y 1]);
18datanorm=~isnan(datanorm(:,1));
19
20%raster = reshape(raster,[x*y z]);
21%img(1:x*y*z/23,1) = int16(0);
22%for i = 1:z/23
23%    img((i)*x*y,1)=1;
24%    img(1+(i-1)*x*y:(i)*x*y,2:23+1) = raster(:,1+(i-1)*23:23*i);
25%end
26%[cell, band] = size(img);
27
28%clear raster;
29
30%nccreate([raster_file,'.nc'],'x','y','band', 'data');
31switch output 
32    case 'txt'
33        input = dlmread(input_file);
34    case 'nc'
35        input = ncread(input_file,'DATA');
36    otherwise
37        disp('Check parameter');
38end
39
40%[ncell, band] = size(input);
41%img(1:x*y,1) = int16(0);
42i = z/23;
43%c = dlmread('sawah_allyears.txt-16.cluster_centres');
44%id= dlmread('sawah_allyears.txt-16.membership');
45%[a,b] = size(id);
46[a,b] = size(input);
47
48idx = a/i;
49hasil(1:idx,1) = 1:idx;
50for n = 1:i
51    hasil(1:idx,n+1) = input(1+(n-1)*idx:idx*n,2);
52end
53
54%%create maps tiff
55[o,p] = size(hasil);
56img=peta_utama(hasil,datanorm);
57img=reshape(img,[x y p]);
58
59geotiffwrite([input_file,'_result.tif'],img(:,:,2:end),R);
60end
